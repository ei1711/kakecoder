int main(){
    int x,y;

    scanf("%d %d",&x,&y);

    int x1,x2,tmp;
    x1=rmax(x,y);
    x2=rmin(x,y);
    int gcd;
    while(1){
        tmp=x1%x2;
        if(tmp==0){
            gcd=x2;
            break;
        }
        x1=x2;
        x2=tmp;
    }
    x/=gcd;
    y/=gcd;

    int cnt=0;
    cnt=x+y;
    //printf("cnt:%d\n",cnt);
    cnt=cnt*gcd-1*(gcd-1);

    printf("%d\n",cnt);
    return 0;
}
