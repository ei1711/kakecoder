int main(){

    for(int i=0;i<h;i++){
        for(int j=0;j<4;j++){
            map[i][j]='0';
        }
    }

        map[h-y[i]-1][x[i]]='1';
    }

    int cnt=0;
    for(int i=0;i<h-1;i++){
        for(int j=0;j<3;j++){
            if(map[i][j]=='0'&&map[i][j+1]=='0'&&map[i+1][j]=='0'&&map[i+1][j+1]=='0'){
                map[i][j]='1';
                map[i][j+1]='1';
                map[i+1][j]='1';
                map[i+1][j+1]='1';
                cnt++;
            }
        }
    }
    printf("%d\n",cnt);
